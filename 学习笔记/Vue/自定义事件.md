# 自定义事件
```html
 <template>
    <div>
        <Student v-on:event_text="GetName" @click.native="event2"></Student>//给组件绑定原生事件需要加native不然会被默认为自定义事件
        上面是通过v-on进行获取数据
        还可以通过ref来获取
        <Student ref="student"></Student>
    </div>
</template>
<script>
    import Student from 'xxx';
    export default {
        name:"App",
        components:{Student},
        methods:{
            GetName(name){
                console.log(name);
            }
        }，
        mounted(){
            this.$refs.student.on('event_text',this.GetName)//通过ref来获取
        },
    }
</script>
```
```html
<template>
    <div>
        <button @click="SendName"></button>
    </div>
</template>
<script>
    //通过父组件给子组件绑定一个自定义事件实现 ：子给父传递数据。
    //z
    export default {
        name:"Student",
        data(){
            return {
                name:"老王"
            }
        },
        methods:{
            SendName(){
                //触发Student组件实例身上的event_texts
                this.$emit('event_text',this.name) 
                //下面是解绑
                this.$off('event_text')//解绑一个
                this.$off('event_text','event_text2')//解绑多个
                this.$off()//解绑全部自定义事件
                
            } 
        }
    }
</script>
```
