# 插槽

  ## 默认插槽

```html
<template>
    <div>
        <Student>
            <span>我将会被放入插槽</span>
        </Student>
    </div>
</template>
<script>
    import Student from 'xxx';
    export default {
        name:"App",
        components:{Student},
        data(){
            return :{
            }
        }
    }
</script>
```
```html
<template>
    <div>
        <button>我是一个按钮</button>
        <slot>我是默认插槽，当用户没有传入时，我会显示</slot>
    </div>
</template>
<script>
    export default {
        name:"Student",
        data(){
            return {
                name:"老王"
            }
        },
//看到102 5：5
    }
</script>
```

## 具名插槽

```html
<template>
    <div>
        <Student>
            <span slot="test1">我将会被放入插槽</span>
            <span slot="test2">我将会被放入插槽2</span>
            <span slot="test2">我将会被放入插槽2</span>
            <template v-slot:test2>//template的特殊写法
                <span>我将会被放入插槽2</span>
            	<span>我将会被放入插槽2</span>
            </template>
        </Student>
    </div>
</template>
<script>
    import Student from 'xxx';
    export default {
        name:"App",
        components:{Student},
        data(){
            return :{
				name:"老王"
            }
        }
    }
</script>
```
```html
<template>
    <div>
        <button>我是一个按钮</button>
        <slot name="test1">我是具名插槽，当用户没有传入时，我会显示</slot>
        <slot name="test2">我是具名插槽，当用户没有传入时，我会显示</slot>
    </div>
</template>
<script>
    export default {
        name:"Student",
        data(){
            return {
                
            }
        },
//看到102 5：5
    }
</script>
```

## 作用域插槽

```html
<template>
    <div>
        <Student>
            //必须包裹template
            <template scope="data">//template的特殊写法
                <span>{{data.name}}</span>
            	<span>{{data.m}}</span>
            </template>
        </Student>
    </div>
</template>
<script>
    import Student from 'xxx';
    export default {
        name:"App",
        components:{Student},
        data(){
            return :{
 				
            }
        }
    }
</script>
```
```html
<template>
    <div>
        <button>我是一个按钮</button>
        <slot :name="name" :msg="哈哈">我是具名插槽，当用户没有传入时，我会显示</slot>
    </div>
</template>
<script>
    export default {
        name:"Student",
        data(){
            return {
               name:"老王"//数据存在于插槽这边的组件中，与前两种插槽相反
            }
        },
    }
</script> 
```
